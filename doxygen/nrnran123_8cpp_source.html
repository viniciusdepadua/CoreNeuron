27 /* Global data structure per process. Using a unique_ptr here causes [minor]
28  * problems because its destructor can be called very late during application
29  * shutdown. If the destructor calls cudaFree and the CUDA runtime has already
30  * been shut down then tools like cuda-memcheck reports errors.
31  */
32 philox4x32_key_t* g_k{};

34 // In a GPU build we need a device-side global pointer to this global state.
35 // This is set to the same unified memory address as `g_k` in
36 // `setup_global_state()` if the GPU is enabled. It would be cleaner to use
37 // __managed__ here, but unfortunately that does not work on machines that do
38 // not have a GPU.
39 #ifdef __CUDACC__
40 CORENRN_DEVICE philox4x32_key_t* g_k_dev;
41 #endif

48 void setup_global_state() {
49  if (g_k) {
50  // Already initialised, nothing to do
51  return;
52  }
53  g_k = coreneuron::allocate_unique<philox4x32_key_t>(
54  coreneuron::unified_allocator<philox4x32_key_t>{})
55  .release();
56 #ifdef __CUDACC__
57  if (coreneuron::unified_memory_enabled()) {
58  // Set the device-side global g_k_dev to point to the newly-allocated
59  // unified memory. If this is false, g_k is just a host pointer and
60  // there is no point initialising the device global to it.
61  {
62  auto const code = cudaMemcpyToSymbol(g_k_dev, &g_k, sizeof(g_k));
63  assert(code == cudaSuccess);
64  }
65  // Make sure g_k_dev is updated.
66  {
67  auto const code = cudaDeviceSynchronize();
68  assert(code == cudaSuccess);
69  }
70  }
71 #endif
72 }

130 CORENRN_HOST_DEVICE uint32_t nrnran123_ipick(nrnran123_State* s) {
131  uint32_t rval;
132  char which = s->which_;
133  rval = s->r.v[which++];
134  if (which > 3) {
135  which = 0;
136  s->c.v[0]++;

139  s->which_ = which;

152 /* at cost of a cached value we could compute two at a time. */
153 CORENRN_HOST_DEVICE double nrnran123_normal(nrnran123_State* s) {
154  double w, x, y;
155  double u1, u2;
156 
157  do {
158  u1 = nrnran123_dblpick(s);
159  u2 = nrnran123_dblpick(s);
160  u1 = 2. * u1 - 1.;
161  u2 = 2. * u2 - 1.;
162  w = (u1 * u1) + (u2 * u2);
163  } while (w > 1);
164 
165  y = std::sqrt((-2. * log(w)) / w);
166  x = u1 * y;
167  return x;
168 }

170 CORENRN_HOST_DEVICE double nrnran123_uint2dbl(uint32_t u) {
171  /* 0 to 2^32-1 transforms to double value in open (0,1) interval */
172  /* min 2.3283064e-10 to max (1 - 2.3283064e-10) */
173  return ((double) u + 1.0) * SHIFT32;
174 }

178  setup_global_state();
179  // If the global seed is changing then we shouldn't have any active streams.
180  {
181  std::lock_guard<OMP_Mutex> _{g_instance_count_mutex};
182  if (g_instance_count != 0 && nrnmpi_myid == 0) {
183  std::cout

188  }
189  }
190  get_global_state().v[0] = gix;

193 /** @brief Allocate a new Random123 stream.
194  * @todo It would be nicer if the API return type was
195  * std::unique_ptr<nrnran123_State, ...not specified...>, so we could use a
196  * custom allocator/deleter and avoid the (fragile) need for matching
197  * nrnran123_deletestream calls.
198  */
199 nrnran123_State* nrnran123_newstream3(uint32_t id1,
200  uint32_t id2,
201  uint32_t id3,
202  bool use_unified_memory) {
203  // The `use_unified_memory` argument is an implementation detail to keep the
204  // old behaviour that some Random123 streams that are known to only be used
205  // from the CPU are allocated using new/delete instead of unified memory.
206  // See OPENACC_EXCLUDED_FILES in coreneuron/CMakeLists.txt. If we dropped
207  // this feature then we could always use coreneuron::unified_allocator.
208 #ifndef CORENEURON_ENABLE_GPU
209  if (use_unified_memory) {
210  throw std::runtime_error("Tried to use CUDA unified memory in a non-GPU build.");
211  }
212 #endif
213  nrnran123_State* s{nullptr};
214  if (use_unified_memory) {
215  s = coreneuron::allocate_unique<nrnran123_State>(
216  coreneuron::unified_allocator<nrnran123_State>{})
217  .release();
218  } else {
219  s = new nrnran123_State{};
220  }
221  s->c.v[0] = 0;
222  s->c.v[1] = id3;
223  s->c.v[2] = id1;
224  s->c.v[3] = id2;
225  nrnran123_setseq(s, 0, 0);
226  {
227  // TODO: can I assert something useful about the instance count going
228  // back to zero anywhere? Or that it is zero when some operations happen?
229  std::lock_guard<OMP_Mutex> _{g_instance_count_mutex};
230  ++g_instance_count;
231  }
232  return s;
233 }
