139 int sparse_thread(SparseObj* so,
140  int n,
141  int* s,
142  int* d,
143  double* t,
144  double dt,
145  SPFUN fun,
146  int linflag,
147  _threadargsproto_) {
148 #define ix(arg) ((arg) *_STRIDE)
149 #define s_(arg) _p[ix(s[arg])]
150 #define d_(arg) _p[ix(d[arg])]
151 
152  int i, j, ierr;
153  double err;
154 
155  for (i = 0; i < n; i++) { /*save old state*/
156  d_(i) = s_(i);
157  }
158  for (err = 1, j = 0; err > CONVERGE; j++) {
159  init_coef_list(so, _iml);
160  spfun(fun, so, so->rhs);
161  if ((ierr = matsol(so, _iml))) {
162  return ierr;
163  }
164  for (err = 0., i = 1; i <= n; i++) { /* why oh why did I write it from 1 */
165  s_(i - 1) += so->rhs[ix(i)];
166  if (!linflag && s_(i - 1) < 0.) {
167  s_(i - 1) = 0.;
168  }
169  err += fabs(so->rhs[ix(i)]);
170  }
171  if (j > MAXSTEPS) {
172  return EXCEED_ITERS;
173  }
174  if (linflag)
175  break;
176  }
177  init_coef_list(so, _iml);
178  spfun(fun, so, so->rhs);
179  for (i = 0; i < n; i++) { /*restore Dstate at t+dt*/
180  d_(i) = (s_(i) - d_(i)) / dt;
181  }
182  return SUCCESS;
183 }

210 static int matsol(SparseObj* so, int _iml) {
211  /* Upper triangularization */
212  so->numop = 0;
213  for (unsigned i = 1; i <= so->neqn; i++) {
214  Elm* pivot;
215  if (fabs((pivot = so->diag[i])->value[_iml]) <= ROUNDOFF) {
216  return SINGULAR;
217  }
218  /* Eliminate all elements in pivot column */
219  for (auto el = pivot->r_down; el; el = el->r_down) {
220  subrow(so, pivot, el, _iml);
221  }
222  }
223  bksub(so, _iml);
224  return (SUCCESS);
225 }

227 static void subrow(SparseObj* so, Elm* pivot, Elm* rowsub, int _iml) {
228  int _cntml_padded = so->_cntml_padded;
229  double r = rowsub->value[_iml] / pivot->value[_iml];
230  so->rhs[ix(rowsub->row)] -= so->rhs[ix(pivot->row)] * r;
231  so->numop++;
232  for (auto el = pivot->c_right; el; el = el->c_right) {
233  for (rowsub = rowsub->c_right; rowsub->col != el->col; rowsub = rowsub->c_right) {
234  ;
235  }
236  rowsub->value[_iml] -= el->value[_iml] * r;
237  so->numop++;
238  }
239 }

241 static void bksub(SparseObj* so, int _iml) {
242  int _cntml_padded = so->_cntml_padded;
243  for (unsigned i = so->neqn; i >= 1; i--) {
244  for (Elm* el = so->diag[i]->c_right; el; el = el->c_right) {
245  so->rhs[ix(el->row)] -= el->value[_iml] * so->rhs[ix(el->col)];
246  so->numop++;
247  }
248  so->rhs[ix(so->diag[i]->row)] /= so->diag[i]->value[_iml];
249  so->numop++;
250  }
251 }

253 static void initeqn(SparseObj* so, unsigned maxeqn) /* reallocate space for matrix */
254 {
255  if (maxeqn == so->neqn)
256  return;
257  free_elm(so);
258  so->neqn = maxeqn;
259  if (so->rowst)
260  Free(so->rowst);
261  if (so->diag)
262  Free(so->diag);
263  if (so->varord)
264  Free(so->varord);
265  if (so->rhs)
266  Free(so->rhs);
267  if (so->ngetcall)
268  free(so->ngetcall);
269  so->elmpool = nullptr;
270  so->rowst = so->diag = nullptr;
271  so->varord = nullptr;
272  so->rowst = (Elm**) myemalloc((maxeqn + 1) * sizeof(Elm*));
273  so->diag = (Elm**) myemalloc((maxeqn + 1) * sizeof(Elm*));
274  so->varord = (unsigned*) myemalloc((maxeqn + 1) * sizeof(unsigned));
275  so->rhs = (double*) myemalloc((maxeqn + 1) * so->_cntml_padded * sizeof(double));
276  so->ngetcall = (unsigned*) ecalloc(so->_cntml_padded, sizeof(unsigned));
277  for (unsigned i = 1; i <= maxeqn; i++) {
278  so->varord[i] = i;
279  so->diag[i] = (Elm*) nrn_pool_alloc(so->elmpool);
280  so->diag[i]->value = (double*) ecalloc(so->_cntml_padded, sizeof(double));
281  so->rowst[i] = so->diag[i];
282  so->diag[i]->row = i;
283  so->diag[i]->col = i;
284  so->diag[i]->r_down = so->diag[i]->r_up = ELM0;
285  so->diag[i]->c_right = so->diag[i]->c_left = ELM0;
286  }
287  unsigned nn = so->neqn * so->_cntml_padded;
288  for (unsigned i = 0; i < nn; ++i) {
289  so->rhs[i] = 0.;
290  }
291 }

308 static Elm* getelm(SparseObj* so, unsigned row, unsigned col, Elm* new_elem)
309 /* return pointer to row col element maintaining order in rows */
310 {
311  Elm *el, *elnext;
312 
313  unsigned vrow = so->varord[row];
314  unsigned vcol = so->varord[col];
315 
316  if (vrow == vcol) {
317  return so->diag[vrow]; /* a common case */
318  }
319  if (vrow > vcol) { /* in the lower triangle */
320  /* search downward from diag[vcol] */
321  for (el = so->diag[vcol];; el = elnext) {
322  elnext = el->r_down;
323  if (!elnext) {
324  break;
325  } else if (elnext->row == row) { /* found it */
326  return elnext;
327  } else if (so->varord[elnext->row] > vrow) {
328  break;
329  }
330  }
331  /* insert below el */
332  if (!new_elem) {
333  new_elem = (Elm*) nrn_pool_alloc(so->elmpool);
334  new_elem->value = (double*) ecalloc(so->_cntml_padded, sizeof(double));
335  increase_order(so, row);
336  }
337  new_elem->r_down = el->r_down;
338  el->r_down = new_elem;
339  new_elem->r_up = el;
340  if (new_elem->r_down) {
341  new_elem->r_down->r_up = new_elem;
342  }
343  /* search leftward from diag[vrow] */
344  for (el = so->diag[vrow];; el = elnext) {
345  elnext = el->c_left;
346  if (!elnext) {
347  break;
348  } else if (so->varord[elnext->col] < vcol) {
349  break;
350  }
351  }
352  /* insert to left of el */
353  new_elem->c_left = el->c_left;
354  el->c_left = new_elem;
355  new_elem->c_right = el;
356  if (new_elem->c_left) {
357  new_elem->c_left->c_right = new_elem;
358  } else {
359  so->rowst[vrow] = new_elem;
360  }
361  } else { /* in the upper triangle */
362  /* search upward from diag[vcol] */
363  for (el = so->diag[vcol];; el = elnext) {
364  elnext = el->r_up;
365  if (!elnext) {
366  break;
367  } else if (elnext->row == row) { /* found it */
368  return elnext;
369  } else if (so->varord[elnext->row] < vrow) {
370  break;
371  }
372  }
373  /* insert above el */
374  if (!new_elem) {
375  new_elem = (Elm*) nrn_pool_alloc(so->elmpool);
376  new_elem->value = (double*) ecalloc(so->_cntml_padded, sizeof(double));
377  increase_order(so, row);
378  }
379  new_elem->r_up = el->r_up;
380  el->r_up = new_elem;
381  new_elem->r_down = el;
382  if (new_elem->r_up) {
383  new_elem->r_up->r_down = new_elem;
384  }
385  /* search right from diag[vrow] */
386  for (el = so->diag[vrow];; el = elnext) {
387  elnext = el->c_right;
388  if (!elnext) {
389  break;
390  } else if (so->varord[elnext->col] > vcol) {
391  break;
392  }
393  }
394  /* insert to right of el */
395  new_elem->c_right = el->c_right;
396  el->c_right = new_elem;
397  new_elem->c_left = el;
398  if (new_elem->c_right) {
399  new_elem->c_right->c_left = new_elem;
400  }
401  }
402  new_elem->row = row;
403  new_elem->col = col;
404  return new_elem;
405 }

447  /* matrix has been set up. Construct the orderlist and orderfind
448  vector.
449  */
450

513 static void get_next_pivot(SparseObj* so, unsigned i) {
514  /* get varord[i], etc. from the head of the orderlist. */
515  Item* order = so->orderlist->next;
516  assert(order != so->orderlist);
517 
518  unsigned j;
519  if ((j = so->varord[order->elm->row]) != i) {
520  /* push order lists down by 1 and put new diag in empty slot */
521  assert(j > i);
522  Elm* el = so->rowst[j];
523  for (; j > i; j--) {
524  so->diag[j] = so->diag[j - 1];
525  so->rowst[j] = so->rowst[j - 1];
526  so->varord[so->diag[j]->row] = j;
527  }
528  so->diag[i] = order->elm;
529  so->rowst[i] = el;
530  so->varord[so->diag[i]->row] = i;
531  /* at this point row links are out of order for diag[i]->col
532  and col links are out of order for diag[i]->row */
533  re_link(so, i);
534  }
535 
536  /* now make sure all needed elements exist */
537  for (Elm* el = so->diag[i]->r_down; el; el = el->r_down) {
538  for (Elm* pivot = so->diag[i]->c_right; pivot; pivot = pivot->c_right) {
539  IGNORE(getelm(so, el->row, pivot->col, ELM0));
540  }
541  reduce_order(so, el->row);
542  }
543 
544 #if CORENRN_DEBUG
545  {
546  int j;
547  Item* _or;
548  printf("%d ", i);
549  for (_or = so->orderlist->next, j = 0; j < 5 && _or != so->orderlist;
550  j++, _or = _or->next) {
551  printf("(%d, %d) ", _or->elm->row, _or->norder);
552  }
553  printf("\n");
554  }
555 #endif
556  delete_item(order);
557 }

643 static void check_assert(SparseObj* so) {
644  /* check that all links are consistent */
645  for (unsigned i = 1; i <= so->neqn; i++) {
646  assert(so->diag[i]);
647  assert(so->diag[i]->row == so->diag[i]->col);
648  assert(so->varord[so->diag[i]->row] == i);
649  assert(so->rowst[i]->row == so->diag[i]->row);

652  assert(el->c_left == ELM0);
653  } else {
654  assert(el->c_left->c_right == el);
655  assert(so->varord[el->c_left->col] < so->varord[el->col]);

669 /* at this point row links are out of order for diag[i]->col
670  and col links are out of order for diag[i]->row */
671 static void re_link(SparseObj* so, unsigned i) {
672  for (Elm* el = so->rowst[i]; el; el = el->c_right) {
673  /* repair hole */
674  if (el->r_up)
675  el->r_up->r_down = el->r_down;
676  if (el->r_down)
677  el->r_down->r_up = el->r_up;
678  }
679 
680  for (Elm* el = so->diag[i]->r_down; el; el = el->r_down) {
681  /* repair hole */
682  if (el->c_right)
683  el->c_right->c_left = el->c_left;
684  if (el->c_left)
685  el->c_left->c_right = el->c_right;
686  else
687  so->rowst[so->varord[el->row]] = el->c_right;
688  }
689 
690  for (Elm* el = so->diag[i]->r_up; el; el = el->r_up) {
691  /* repair hole */
692  if (el->c_right)
693  el->c_right->c_left = el->c_left;
694  if (el->c_left)
695  el->c_left->c_right = el->c_right;
696  else
697  so->rowst[so->varord[el->row]] = el->c_right;
698  }
699 
700  /* matrix is consistent except that diagonal row elements are unlinked from
701  their columns and the diagonal column elements are unlinked from their
702  rows.
703  For simplicity discard all knowledge of links and use getelm to relink
704  */
705  Elm *dright, *dleft, *dup, *ddown, *elnext;
706 
707  so->rowst[i] = so->diag[i];
708  dright = so->diag[i]->c_right;
709  dleft = so->diag[i]->c_left;
710  dup = so->diag[i]->r_up;
711  ddown = so->diag[i]->r_down;
712  so->diag[i]->c_right = so->diag[i]->c_left = ELM0;
713  so->diag[i]->r_up = so->diag[i]->r_down = ELM0;
714  for (Elm* el = dright; el; el = elnext) {
715  elnext = el->c_right;
716  IGNORE(getelm(so, el->row, el->col, el));
717  }
718  for (Elm* el = dleft; el; el = elnext) {
719  elnext = el->c_left;
720  IGNORE(getelm(so, el->row, el->col, el));
721  }
722  for (Elm* el = dup; el; el = elnext) {
723  elnext = el->r_up;
724  IGNORE(getelm(so, el->row, el->col, el));
725  }
726  for (Elm* el = ddown; el; el = elnext) {
727  elnext = el->r_down;
728  IGNORE(getelm(so, el->row, el->col, el));
729  }
730 }
