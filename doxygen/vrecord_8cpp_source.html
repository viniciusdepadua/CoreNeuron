115 double VecPlayContinuous::interpolate(double tt) {
116  if (tt >= t_[ubound_index_]) {
117  last_index_ = ubound_index_;
118  if (last_index_ == 0) {
119  // printf("return last tt=%g ubound=%g y=%g\n", tt, t_->elem(ubound_index_),
120  // y_->elem(last_index_));
121  return y_[last_index_];
122  }
123  } else if (tt <= t_[0]) {
124  last_index_ = 0;
125  // printf("return elem(0) tt=%g t0=%g y=%g\n", tt, t_->elem(0), y_->elem(0));
126  return y_[0];
127  } else {
128  search(tt);
129  }
130  double x0 = y_[last_index_ - 1];
131  double x1 = y_[last_index_];
132  double t0 = t_[last_index_ - 1];
133  double t1 = t_[last_index_];
134  // printf("IvocVectRecorder::continuous tt=%g t0=%g t1=%g theta=%g x0=%g x1=%g\n", tt, t0, t1,
135  // (tt - t0)/(t1 - t0), x0, x1);
136  if (t0 == t1) {
137  return (x0 + x1) / 2.;
138  }
139  return interp((tt - t0) / (t1 - t0), x0, x1);
140 }
