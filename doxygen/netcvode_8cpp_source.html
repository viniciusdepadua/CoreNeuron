80 void net_send(void** v, int weight_index_, Point_process* pnt, double td, double flag) {
81  NrnThread* nt = PP2NT(pnt);
82  NetCvodeThreadData& p = net_cvode_instance->p[nt->id];
83  SelfEvent* se = new SelfEvent;
84  se->flag_ = flag;
85  se->target_ = pnt;
86  se->weight_index_ = weight_index_;
87  if (v >= nt->_vdata) {
88  se->movable_ = v; // needed for SaveState
89  }
90  assert(net_cvode_instance);
91  ++p.unreffed_event_cnt_;
92  if (td < nt->_t) {
93  char buf[100];
94  sprintf(buf, "net_send td-t = %g", td - nt->_t);
95  se->pr(buf, td, net_cvode_instance);
96  abort();
97  hoc_execerror("net_send delay < 0", 0);
98  }
99  TQItem* q = net_cvode_instance->event(td, se, nt);
100  if (flag == 1.0 && v >= nt->_vdata) {
101  *v = (void*) q;
102  }
103  // printf("net_send %g %s %g %p\n", td, pnt_name(pnt), flag, *v);
104 }

241  p[i].tqe_->nshift_ = -1;
242  p[i].tqe_->shift_bin(nrn_threads->_t);

524 void NetCvode::check_thresh(NrnThread* nt) { // for default method
525  Instrumentor::phase p("check-threshold");
526  double teps = 1e-10;
527 
528  nt->_net_send_buffer_cnt = 0;
529  int net_send_buf_count = 0;
530  PreSyn* presyns = nt->presyns;
531  PreSynHelper* presyns_helper = nt->presyns_helper;
532  double* actual_v = nt->_actual_v;
533 
534 #if defined(_OPENACC)
535  int stream_id = nt->stream_id;
536 #endif
537 
538  if (nt->ncell == 0)
539  return;
540 
541  //_net_send_buffer_cnt is no longer used in openacc kernel, remove this?
542  //#ifdef _OPENACC
543  // if(nt->compute_gpu)
544  // acc_update_device(&(nt->_net_send_buffer_cnt), sizeof(int));
545  //#endif
546 
547  // on GPU...
548  // clang-format off
549 
550  #pragma acc parallel loop present( \
551  nt[0:1], presyns_helper[0:nt->n_presyn], \
552  presyns[0:nt->n_presyn], actual_v[0:nt->end]) \
553  copy(net_send_buf_count) if (nt->compute_gpu) \
554  async(stream_id)
555  // clang-format on
556  for (int i = 0; i < nt->ncell; ++i) {
557  PreSyn* ps = presyns + i;
558  PreSynHelper* psh = presyns_helper + i;
559  int idx = 0;
560  int thidx = ps->thvar_index_;
561  double v = actual_v[thidx];
562  double threshold = ps->threshold_;
563  int* flag = &(psh->flag_);
564 
565  if (pscheck(v, threshold, flag)) {
566 #ifndef _OPENACC
567  nt->_net_send_buffer_cnt = net_send_buf_count;
568  if (nt->_net_send_buffer_cnt >= nt->_net_send_buffer_size) {
569  nt->_net_send_buffer_size *= 2;
570  nt->_net_send_buffer = (int*) erealloc(nt->_net_send_buffer,
571  nt->_net_send_buffer_size * sizeof(int));
572  }
573 #endif
574 
575  // clang-format off
576 
577  #pragma acc atomic capture
578  // clang-format on
579  idx = net_send_buf_count++;
580 
581  nt->_net_send_buffer[idx] = i;
582  }
583  }
584 
585  // clang-format off
586 
587  #pragma acc wait(stream_id)
588  // clang-format on
589  nt->_net_send_buffer_cnt = net_send_buf_count;
590 
591  if (nt->_net_send_buffer_cnt) {
592 #ifdef _OPENACC
593  int* nsbuffer = nt->_net_send_buffer;
594 #endif
595  // clang-format off
596 
597  #pragma acc update host(nsbuffer[0:nt->_net_send_buffer_cnt]) if (nt->compute_gpu) async(stream_id)
598  #pragma acc wait(stream_id)
599  // clang-format on
600  }
601 
602  // on CPU...
603  for (int i = 0; i < nt->_net_send_buffer_cnt; ++i) {
604  PreSyn* ps = nt->presyns + nt->_net_send_buffer[i];
605  ps->send(nt->_t + teps, net_cvode_instance, nt);
606  }
607 
608  // Types that have WATCH statements. If exist, then last element is 0.
609  if (nt->_watch_types) {
610  for (int i = 0; nt->_watch_types[i] != 0; ++i) {
611  int type = nt->_watch_types[i];
612  (*corenrn.get_watch_check()[type])(nt, nt->_ml_list[type]);
613  // may generate net_send events (with 0 (teps) delay)
614  }
615  }
616 }

618 // WATCH statements are rare. Conceptually they are very similar to
619 // PreSyn thresholds as above but an optimal peformance implementation for GPU is
620 // not obvious. Each WATCH statement threshold test could make use of
621 // pscheck. Note that it is possible that there are several active WATCH
622 // statements for a given POINT_PROCESS instance as well as none active.
623 // Also WATCH statements switch between active and inactive state.
624 //
625 // In NEURON,
626 // both PreSyn and WatchCondition were subclasses of ConditionEvent. When
627 // a WatchCondition fired in the fixed step method, it was placed on the queue
628 // with a delivery time of t+teps. WatchCondition::deliver called the NET_RECEIVE
629 // block with proper flag ( but nullptr weight vector). WatchConditions
630 // were created,added/removed,destroyed from a list as necessary.
631 // Perhaps the most commonly used WATCH statement is in the context of a
632 // ThresholdDetect Point_process which watches voltage and compares to
633 // an instance specific threshold parameter. A firing ThresholdDetect instance
634 // would call net_event(tdeliver) which then feeds into the standard
635 // artcell PreSyn sequence (using pntsrc_ instead of thvar_index_).
636 //
637 // So... the PreSyns have the same order as they are checked (although PreSyn
638 // data is AoS instead of SoA and nested 'if' means a failure of SIMD.)
639 // But if multiple WATCH, there is (from one kind of implementation viewpoint),
640 // yet another 'if' with regard to whether a WATCH is active. And if there
641 // are multiple WATCH, the size of the list is dynamic.
642 //
643 // An experimental implementation is to check all WATCH of all instances
644 // of a type with the proviso that there is an active flag for each WATCH.
645 // ie. active, below, var1, var2 are all SoA (except one of the var may
646 // be voltage). Can use 'if (active && pscheck(var1, var2, &below)'
647 // The mod file net_send_buffering fragments can be used which
648 // ultimately call net_send using a transient SelfEvent. ie. all
649 // checking computation takes place in the context of the mod file without
650 // using explicit WatchCondition instances.
651 
652 // events including binqueue events up to t+dt/2
653 void NetCvode::deliver_net_events(NrnThread* nt) { // for default method
654 #if NRN_MULTISEND
655  if (use_multisend_ && nt->id == 0) {
656  nrn_multisend_advance();
657  }
658 #endif
659  int tid = nt->id;
660  double tsav = nt->_t;
661  double tm = nt->_t + 0.5 * nt->_dt;
662 tryagain:
663  // one of the events on the main queue may be a NetParEvent
664  // which due to dt round off error can result in an event
665  // placed on the bin queue to be delivered now, which
666  // can put 0 delay events on to the main queue. So loop til
667  // no events. The alternative would be to deliver an idt=0 event
668  // immediately but that would very much change the sequence
669  // with respect to what is being done here and it is unclear
670  // how to fix the value of t there. This can be a do while loop
671  // but I do not want to affect the case of not using a bin queue.
672 
673  if (nrn_use_bin_queue_) {
674  TQItem* q;
675  while ((q = p[tid].tqe_->dequeue_bin()) != 0) {
676  DiscreteEvent* db = (DiscreteEvent*) q->data_;
677 
678 #if PRINT_EVENT
679  if (print_event_) {
680  db->pr("binq deliver", nrn_threads->_t, this);
681  }
682 #endif
683 
684  delete q;
685  db->deliver(nt->_t, this, nt);
686  }
687  // assert(int(tm/nt->_dt)%1000 == p[tid].tqe_->nshift_);
688  }
689 
690  deliver_events(tm, nt);
691 
692  if (nrn_use_bin_queue_) {
693  if (p[tid].tqe_->top()) {
694  goto tryagain;
695  }
696  p[tid].tqe_->shift_bin(tm);
697  }
698 
699  nt->_t = tsav;
700 
701  /*before executing on gpu, we have to update the NetReceiveBuffer_t on GPU */
702  update_net_receive_buffer(nt);
703 
704  for (auto& net_buf_receive: corenrn.get_net_buf_receive()) {
705  std::string ss("net-buf-receive-");
706  ss += nrn_get_mechname(net_buf_receive.second);
707  Instrumentor::phase p_net_buf_receive(ss.c_str());
708  (*net_buf_receive.first)(nt);
709  }
710 }
