165 void corenrn_parameters::parse(int argc, char** argv) {
166  try {
167  app.parse(argc, argv);
168  if (verbose == verbose_level::NONE) {
169  nrn_nobanner_ = 1;
170  }
171  } catch (const CLI::ExtrasError& e) {
172  // in case of parsing errors, show message with exception
173  std::cerr << "CLI parsing error, see nrniv-core --help for more information. \n"
174  << std::endl;
175  app.exit(e);
176  throw e;
177 
178  } catch (const CLI::ParseError& e) {
179  // use --help is also ParseError; in this case exit by showing all options
180  app.exit(e);
181  exit(0);
182  }
183 
184  // is user has asked for version info, print it and exit
185  if (show_version) {
186  std::cout << "CoreNEURON Version : " << cnrn_version() << std::endl;
187  exit(0);
188  }
