16 void ReportHandler::create_report(double dt, double tstop, double delay) {
17 #if defined(ENABLE_BIN_REPORTS) || defined(ENABLE_SONATA_REPORTS)
18  if (m_report_config.start < t) {
19  m_report_config.start = t;
20  }
21  m_report_config.stop = std::min(m_report_config.stop, tstop);
22 
23  for (const auto& mech: m_report_config.mech_names) {
24  m_report_config.mech_ids.emplace_back(nrn_get_mechtype(mech.data()));
25  }
26  if (m_report_config.type == SynapseReport && m_report_config.mech_ids.empty()) {
27  std::cerr << "[ERROR] mechanism to report: " << m_report_config.mech_names[0]
28  << " is not mapped in this simulation, cannot report on it \n";
29  nrn_abort(1);
30  }
31  for (int ith = 0; ith < nrn_nthread; ++ith) {
32  NrnThread& nt = nrn_threads[ith];
33  double* report_variable = nt._actual_v;
34  if (!nt.ncell) {
35  continue;
36  }
37  const std::vector<int>& nodes_to_gid = map_gids(nt);
38  VarsToReport vars_to_report;
39  bool is_soma_target;
40  switch (m_report_config.type) {
41  case IMembraneReport:
42  report_variable = nt.nrn_fast_imem->nrn_sav_rhs;
43  case SectionReport:
44  vars_to_report =
45  get_section_vars_to_report(nt,
46  m_report_config.target,
47  report_variable,
48  m_report_config.section_type,
49  m_report_config.section_all_compartments);
50  is_soma_target = m_report_config.section_type == SectionType::Soma ||
51  m_report_config.section_type == SectionType::Cell;
52  register_section_report(nt, m_report_config, vars_to_report, is_soma_target);
53  break;
54  case SummationReport:
55  vars_to_report = get_summation_vars_to_report(nt,
56  m_report_config.target,
57  m_report_config,
58  nodes_to_gid);
59  register_custom_report(nt, m_report_config, vars_to_report);
60  break;
61  default:
62  vars_to_report = get_synapse_vars_to_report(nt, m_report_config, nodes_to_gid);
63  register_custom_report(nt, m_report_config, vars_to_report);
64  }
65  if (!vars_to_report.empty()) {
66  auto report_event = std::make_unique<ReportEvent>(dt,
67  t,
68  vars_to_report,
69  m_report_config.output_path.data(),
70  m_report_config.report_dt);
71  report_event->send(t, net_cvode_instance, &nt);
72  m_report_events.push_back(std::move(report_event));
73  }
74  }
75 #else
76  if (nrnmpi_myid == 0) {
77  std::cerr << "[WARNING] : Reporting is disabled. Please recompile with either libsonata or "
78  "reportinglib. \n";
79  }
80 #endif // defined(ENABLE_BIN_REPORTS) || defined(ENABLE_SONATA_REPORTS)
81 }

121 void register_sections_to_report(const SecMapping* sections,
122  std::vector<VarWithMapping>& to_report,
123  double* report_variable,
124  bool all_compartments) {
125  for (const auto& section: sections->secmap) {
126  // compartment_id
127  int section_id = section.first;
128  const auto& segment_ids = section.second;
129 
130  // get all compartment values (otherwise, just middle point)
131  if (all_compartments) {
132  for (const auto& segment_id: segment_ids) {
133  // corresponding voltage in coreneuron voltage array
134  double* variable = report_variable + segment_id;
135  to_report.emplace_back(VarWithMapping(section_id, variable));
136  }
137  } else {
138  nrn_assert(segment_ids.size() % 2);
139  // corresponding voltage in coreneuron voltage array
140  const auto segment_id = segment_ids[segment_ids.size() / 2];
141  double* variable = report_variable + segment_id;
142  to_report.emplace_back(VarWithMapping(section_id, variable));
143  }
144  }

147 VarsToReport ReportHandler::get_section_vars_to_report(const NrnThread& nt,
148  const std::set<int>& target,
149  double* report_variable,
150  SectionType section_type,
151  bool all_compartments) const {
152  VarsToReport vars_to_report;
153  const auto& section_type_str = getSectionTypeStr(section_type);
154  const auto* mapinfo = static_cast<NrnThreadMappingInfo*>(nt.mapping);
155  if (!mapinfo) {
156  std::cerr << "[COMPARTMENTS] Error : mapping information is missing for a Cell group "
157  << nt.ncell << '\n';
158  nrn_abort(1);
159  }
160 
161  for (int i = 0; i < nt.ncell; i++) {
162  int gid = nt.presyns[i].gid_;
163  if (target.find(gid) != target.end()) {
164  const auto& cell_mapping = mapinfo->get_cell_mapping(gid);
165  if (cell_mapping == nullptr) {
166  std::cerr
167  << "[COMPARTMENTS] Error : Compartment mapping information is missing for gid "
168  << gid << '\n';
169  nrn_abort(1);
170  }
171  std::vector<VarWithMapping> to_report;
172  to_report.reserve(cell_mapping->size());
173 
174  if (section_type_str == "All") {
175  const auto& section_mapping = cell_mapping->secmapvec;
176  for (const auto& sections: section_mapping) {
177  register_sections_to_report(sections,
178  to_report,
179  report_variable,
180  all_compartments);
181  }
182  } else {
183  /** get section list mapping for the type, if available */
184  if (cell_mapping->get_seclist_section_count(section_type_str) > 0) {
185  const auto& sections = cell_mapping->get_seclist_mapping(section_type_str);
186  register_sections_to_report(sections,
187  to_report,
188  report_variable,
189  all_compartments);
190  }
191  }
192  vars_to_report[gid] = to_report;
193  }
194  }
195  return vars_to_report;
196 }

198 VarsToReport ReportHandler::get_summation_vars_to_report(
199  const NrnThread& nt,
200  const std::set<int>& target,
201  ReportConfiguration& report,
202  const std::vector<int>& nodes_to_gids) const {
203  VarsToReport vars_to_report;
204  const auto* mapinfo = static_cast<NrnThreadMappingInfo*>(nt.mapping);
205  auto& summation_report = nt.summation_report_handler_->summation_reports_[report.output_path];
206  if (!mapinfo) {
207  std::cerr << "[COMPARTMENTS] Error : mapping information is missing for a Cell group "
208  << nt.ncell << '\n';
209  nrn_abort(1);
210  }
211 
212  for (int i = 0; i < nt.ncell; i++) {
213  int gid = nt.presyns[i].gid_;
214  if (report.target.find(gid) == report.target.end()) {
215  continue;
216  }
217  bool has_imembrane = false;
218  // In case we need convertion of units
219  int scale = 1;
220  for (auto i = 0; i < report.mech_ids.size(); ++i) {
221  auto mech_id = report.mech_ids[i];
222  auto var_name = report.var_names[i];
223  auto mech_name = report.mech_names[i];
224  if (mech_name != "i_membrane") {
225  // need special handling for Clamp processes to flip the current value
226  if (mech_name == "IClamp" || mech_name == "SEClamp") {
227  scale = -1;
228  }
229  Memb_list* ml = nt._ml_list[mech_id];
230  if (!ml) {
231  continue;
232  }
233 
234  for (int j = 0; j < ml->nodecount; j++) {
235  auto segment_id = ml->nodeindices[j];
236  if ((nodes_to_gids[ml->nodeindices[j]] == gid)) {
237  double* var_value =
238  get_var_location_from_var_name(mech_id, var_name.data(), ml, j);
239  summation_report.currents_[segment_id].push_back(
240  std::make_pair(var_value, scale));
241  }
242  }
243  } else {
244  has_imembrane = true;
245  }
246  }
247  if (target.find(gid) != target.end()) {
248  const auto& cell_mapping = mapinfo->get_cell_mapping(gid);
249  if (cell_mapping == nullptr) {
250  std::cerr
251  << "[SUMMATION] Error : Compartment mapping information is missing for gid "
252  << gid << '\n';
253  nrn_abort(1);
254  }
255  std::vector<VarWithMapping> to_report;
256  to_report.reserve(cell_mapping->size());
257  summation_report.summation_.resize(nt.end);
258  double* report_variable = summation_report.summation_.data();
259  const auto& section_type_str = getSectionTypeStr(report.section_type);
260  if (report.section_type != SectionType::All) {
261  if (cell_mapping->get_seclist_section_count(section_type_str) > 0) {
262  const auto& sections = cell_mapping->get_seclist_mapping(section_type_str);
263  register_sections_to_report(sections,
264  to_report,
265  report_variable,
266  report.section_all_compartments);
267  }
268  }
269  const auto& section_mapping = cell_mapping->secmapvec;
270  for (const auto& sections: section_mapping) {
271  for (auto& section: sections->secmap) {
272  // compartment_id
273  int section_id = section.first;
274  auto& segment_ids = section.second;
275  for (const auto& segment_id: segment_ids) {
276  // corresponding voltage in coreneuron voltage array
277  if (has_imembrane) {
278  summation_report.currents_[segment_id].push_back(
279  std::make_pair(nt.nrn_fast_imem->nrn_sav_rhs + segment_id, 1));
280  }
281  if (report.section_type == SectionType::All) {
282  double* variable = report_variable + segment_id;
283  to_report.emplace_back(VarWithMapping(section_id, variable));
284  } else if (report.section_type == SectionType::Cell) {
285  summation_report.gid_segments_[gid].push_back(segment_id);
286  }
287  }
288  }
289  }
290  vars_to_report[gid] = to_report;
291  }
292  }
293  return vars_to_report;
294 }

296 VarsToReport ReportHandler::get_synapse_vars_to_report(
297  const NrnThread& nt,
298  ReportConfiguration& report,
299  const std::vector<int>& nodes_to_gids) const {
300  VarsToReport vars_to_report;
301  for (int i = 0; i < nt.ncell; i++) {
302  int gid = nt.presyns[i].gid_;
303  if (report.target.find(gid) == report.target.end()) {
304  continue;
305  }
306  // There can only be 1 mechanism
307  nrn_assert(report.mech_ids.size() == 1);
308  auto mech_id = report.mech_ids[0];
309  auto var_name = report.var_names[0];
310  Memb_list* ml = nt._ml_list[mech_id];
311  if (!ml) {
312  continue;
313  }
314  std::vector<VarWithMapping> to_report;
315  to_report.reserve(ml->nodecount);
316 
317  for (int j = 0; j < ml->nodecount; j++) {
318  double* is_selected =
319  get_var_location_from_var_name(mech_id, SELECTED_VAR_MOD_NAME, ml, j);
320  bool report_variable = false;
321 
322  /// if there is no variable in mod file then report on every compartment
323  /// otherwise check the flag set in mod file
324  if (is_selected == nullptr) {
325  report_variable = true;
326  } else {
327  report_variable = *is_selected != 0.;
328  }
329  if ((nodes_to_gids[ml->nodeindices[j]] == gid) && report_variable) {
330  double* var_value = get_var_location_from_var_name(mech_id, var_name.data(), ml, j);
331  double* synapse_id =
332  get_var_location_from_var_name(mech_id, SYNAPSE_ID_MOD_NAME, ml, j);
333  nrn_assert(synapse_id && var_value);
334  to_report.emplace_back(static_cast<int>(*synapse_id), var_value);
335  }
336  }
337  if (!to_report.empty()) {
338  vars_to_report[gid] = to_report;
339  }
340  }
341  return vars_to_report;
342 }

344 // map GIDs of every compartment, it consist in a backward sweep then forward sweep algorithm
345 std::vector<int> ReportHandler::map_gids(const NrnThread& nt) const {
346  std::vector<int> nodes_gid(nt.end, -1);
347  // backward sweep: from presyn compartment propagate back GID to parent
348  for (int i = 0; i < nt.n_presyn; i++) {
349  const int gid = nt.presyns[i].gid_;
350  const int thvar_index = nt.presyns[i].thvar_index_;
351  // only for non artificial cells
352  if (thvar_index >= 0) {
353  // setting all roots gids of the presyns nodes,
354  // index 0 have parent set to 0, so we must stop at j > 0
355  // also 0 is the parent of all, so it is an error to attribute a GID to it.
356  nodes_gid[thvar_index] = gid;
357  for (int j = thvar_index; j > 0; j = nt._v_parent_index[j]) {
358  nodes_gid[nt._v_parent_index[j]] = gid;
359  }
360  }
361  }
362  // forward sweep: setting all compartements nodes to the GID of its root
363  // already sets on above loop. This is working only because compartments are stored in order
364  // parents follow by childrens
365  for (int i = nt.ncell + 1; i < nt.end; i++) {
366  nodes_gid[i] = nodes_gid[nt._v_parent_index[i]];
367  }
368  return nodes_gid;
369 }
