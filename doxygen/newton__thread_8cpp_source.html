62 int nrn_newton_thread(NewtonSpace* ns,
63  int n,
64  int* s,
65  NEWTFUN pfunc,
66  double* value,
67  _threadargsproto_) {
68  int count = 0, error = 0;
69  double change = 1.0, max_dev, temp;
70  int done = 0;
71  /*
72  * Create arrays for Jacobian, variable increments, function values, and
73  * permutation vector
74  */
75  double* delta_x = ns->delta_x;
76  double** jacobian = ns->jacobian;
77  int* perm = ns->perm;
78  /* Iteration loop */
79  while (!done) {
80  if (count++ >= MAXITERS) {
81  error = EXCEED_ITERS;
82  done = 2;
83  }
84  if (!done && change > MAXCHANGE) {
85  /*
86  * Recalculate Jacobian matrix if solution has changed by more
87  * than MAXCHANGE
88  */
89 
90  nrn_buildjacobian_thread(ns, n, s, pfunc, value, jacobian, _threadargs_);
91  for (int i = 0; i < n; i++)
92  value[ix(i)] = -value[ix(i)]; /* Required correction to
93  * function values */
94  error = nrn_crout_thread(ns, n, jacobian, perm, _threadargs_);
95  if (error != SUCCESS) {
96  done = 2;
97  }
98  }
99 
100  if (!done) {
101  nrn_scopmath_solve_thread(n, jacobian, value, perm, delta_x, (int*) 0, _threadargs_);
102 
103  /* Update solution vector and compute norms of delta_x and value */
104 
105  change = 0.0;
106  if (s) {
107  for (int i = 0; i < n; i++) {
108  if (fabs(s_(i)) > ZERO && (temp = fabs(delta_x[ix(i)] / (s_(i)))) > change)
109  change = temp;
110  s_(i) += delta_x[ix(i)];
111  }
112  } else {
113  for (int i = 0; i < n; i++) {
114  if (fabs(s_(i)) > ZERO && (temp = fabs(delta_x[ix(i)] / (s_(i)))) > change)
115  change = temp;
116  s_(i) += delta_x[ix(i)];
117  }
118  }
119  newtfun(pfunc); /* Evaluate function values with new solution */
120  max_dev = 0.0;
121  for (int i = 0; i < n; i++) {
122  value[ix(i)] = -value[ix(i)]; /* Required correction to function
123  * values */
124  if ((temp = fabs(value[ix(i)])) > max_dev)
125  max_dev = temp;
126  }
127 
128  /* Check for convergence or maximum iterations */
129 
130  if (change <= CONVERGE && max_dev <= ZERO) {
131  // break;
132  done = 1;
133  }
134  }
135  } /* end of while loop */
136 
137  return (error);
138 }

140 /*------------------------------------------------------------*/
141 /* */
142 /* BUILDJACOBIAN */
143 /* */
144 /* Creates the Jacobian matrix by computing partial deriv- */
145 /* atives by finite central differences. If the column */
146 /* variable is nonzero, an increment of 2% of the variable */
147 /* is used. STEP is the minimum increment allowed; it is */
148 /* currently set to 1.0E-6. */
149 /* */
150 /* Returns: no return variable */
151 /* */
152 /* Calling sequence: */
153 /* buildjacobian(n, index, x, pfunc, value, jacobian) */
154 /* */
155 /* Arguments: */
156 /* */
157 /* Input: n, integer, number of variables */
158 /* */
159 /* x, pointer to array of addresses of the solution */
160 /* vector elements */
161 /* */
162 /* p, array of parameter values */
163 /* */
164 /* pfunc, pointer to function which computes the */
165 /* deviation from zero of each equation */
166 /* in the model. */
167 /* */
168 /* value, pointer to array of addresses of function */
169 /* values */
170 /* */
171 /* Output: jacobian, double, computed jacobian matrix */
172 /* */
173 /* Functions called: user-supplied function with argument */
174 /* (p) to compute vector of function */
175 /* values for each equation. */
176 /* makevector(), freevector() */
177 /* */
178 /*------------------------------------------------------------*/

180 #define max(x, y) (fabs(x) > y ? x : y)

182 void nrn_buildjacobian_thread(NewtonSpace* ns,
183  int n,
184  int* index,
185  NEWTFUN pfunc,
186  double* value,
187  double** jacobian,
188  _threadargsproto_) {
189 #define x_(arg) _p[(arg) *_STRIDE]
190  double* high_value = ns->high_value;
191  double* low_value = ns->low_value;
192 
193  /* Compute partial derivatives by central finite differences */
194 
195  for (int j = 0; j < n; j++) {
196  double increment = max(fabs(0.02 * (x_(index[j]))), STEP);
197  x_(index[j]) += increment;
198  newtfun(pfunc);
199  for (int i = 0; i < n; i++)
200  high_value[ix(i)] = value[ix(i)];
201  x_(index[j]) -= 2.0 * increment;
202  newtfun(pfunc);
203  for (int i = 0; i < n; i++) {
204  low_value[ix(i)] = value[ix(i)];
205 
206  /* Insert partials into jth column of Jacobian matrix */
207 
208  jacobian[i][ix(j)] = (high_value[ix(i)] - low_value[ix(i)]) / (2.0 * increment);
209  }
210 
211  /* Restore original variable and function values. */
212 
213  x_(index[j]) += increment;
214  newtfun(pfunc);
215  }
216 }
