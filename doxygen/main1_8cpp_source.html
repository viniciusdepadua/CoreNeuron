83  * Convert char* containing arguments from neuron to char* argv[] for

86 char* prepare_args(int& argc, char**& argv, int use_mpi, const char* mpi_lib, const char* arg) {
87  // first construct all arguments as string
88  std::string args(arg);
89  args.insert(0, " coreneuron ");
90  args.append(" --skip-mpi-finalize ");
91  if (use_mpi) {
92  args.append(" --mpi ");
93  }
94 
95  // if neuron has passed name of MPI library then add it to CLI
96  std::string corenrn_mpi_lib{mpi_lib};
97  if (!corenrn_mpi_lib.empty()) {
98  args.append(" --mpi-lib ");
99  corenrn_mpi_lib += " ";
100  args.append(corenrn_mpi_lib);
101  }
102 
103  // we can't modify string with strtok, make copy
104  char* first = strdup(args.c_str());
105  const char* sep = " ";
106 
107  // first count the no of argument
108  char* token = strtok(first, sep);
109  argc = 0;
110  while (token) {
111  token = strtok(nullptr, sep);
112  argc++;
113  }
114  free(first);
115 
116  // now build char*argv
117  argv = new char*[argc];
118  first = strdup(args.c_str());
119  token = strtok(first, sep);
120  for (int i = 0; token; i++) {
121  argv[i] = token;
122  token = strtok(nullptr, sep);
123  }
124 
125  // return actual data to be freed
126  return first;
127 }

133 // bsize = 0 then per step transfer
134 // bsize > 1 then full trajectory save into arrays.
135 void get_nrn_trajectory_requests(int bsize) {
136  if (nrn2core_get_trajectory_requests_) {
137  for (int tid = 0; tid < nrn_nthread; ++tid) {
138  NrnThread& nt = nrn_threads[tid];
139  int n_pr;
140  int n_trajec;
141  int* types;
142  int* indices;
143  void** vpr;
144  double** varrays;
145  double** pvars;
146 
147  // bsize is passed by reference, the return value will determine if
148  // per step return or entire trajectory return.
149  (*nrn2core_get_trajectory_requests_)(
150  tid, bsize, n_pr, vpr, n_trajec, types, indices, pvars, varrays);
151  delete_trajectory_requests(nt);
152  if (n_trajec) {
153  TrajectoryRequests* tr = new TrajectoryRequests;
154  nt.trajec_requests = tr;
155  tr->bsize = bsize;
156  tr->n_pr = n_pr;
157  tr->n_trajec = n_trajec;
158  tr->vsize = 0;
159  tr->vpr = vpr;
160  tr->gather = new double*[n_trajec];
161  tr->varrays = varrays;
162  tr->scatter = pvars;
163  for (int i = 0; i < n_trajec; ++i) {
164  tr->gather[i] = stdindex2ptr(types[i], indices[i], nt);
165  }
166  delete[] types;
167  delete[] indices;
168  }
169  }
170  }

174  char* argv[],

183  Instrumentor::stop_profile();

186  if (!corenrn_param.is_quiet()) {
187  report_mem_usage("After MPI_Init");

189 
190  // initialise default coreneuron parameters
191  initnrn();
192 
193  // set global variables
194  // precedence is: set by user, globals.dat, 34.0
195  celsius = corenrn_param.celsius;
196 
197 #if _OPENACC
198  if (!corenrn_param.gpu && corenrn_param.cell_interleave_permute == 2) {
199  fprintf(stderr,
200  "compiled with _OPENACC does not allow the combination of --cell-permute=2 and "
201  "missing --gpu\n");
202  exit(1);
203  }
204  if (!corenrn_param.gpu && corenrn_param.cuda_interface) {
205  fprintf(stderr,
206  "compiled with OpenACC/CUDA does not allow the combination of --cuda-interface and "
207  "missing --gpu\n");
208  exit(1);
209  }
210 #endif
211 
212 // if multi-threading enabled, make sure mpi library supports it
213 #if NRNMPI
214  if (corenrn_param.mpi_enable && corenrn_param.threading) {
215  nrnmpi_check_threading_support();
216  }
217 #endif
218 
219  // full path of files.dat file
220  std::string filesdat(corenrn_param.datpath + "/" + corenrn_param.filesdat);
221 
222  // read the global variable names and set their values from globals.dat
223  set_globals(corenrn_param.datpath.c_str(), (corenrn_param.seed >= 0), corenrn_param.seed);
224 
225  // set global variables for start time, timestep and temperature
226  if (!corenrn_embedded) {
227  t = checkPoints.restore_time();
228  }
229 
230  if (corenrn_param.dt != -1000.) { // command line arg highest precedence
231  dt = corenrn_param.dt;
232  } else if (dt == -1000.) { // not on command line and no dt in globals.dat
233  dt = 0.025; // lowest precedence
234  }
235 
236  corenrn_param.dt = dt;
237 
238  rev_dt = (int) (1. / dt);
239 
240  if (corenrn_param.celsius != -1000.) { // command line arg highest precedence
241  celsius = corenrn_param.celsius;
242  } else if (celsius == -1000.) { // not on command line and no celsius in globals.dat
243  celsius = 34.0; // lowest precedence
244  }
245 
246  corenrn_param.celsius = celsius;
247 
248  // for ispc backend
249  ispc_celsius = celsius;
250 
251  // create net_cvode instance
252  mk_netcvode();
253 
254  // One part done before call to nrn_setup. Other part after.
255 
256  if (!corenrn_param.patternstim.empty()) {
257  nrn_set_extra_thread0_vdata();
258  }
259 
260  if (!corenrn_param.is_quiet()) {
261  report_mem_usage("Before nrn_setup");
262  }
263 
264  // set if need to interleave cells
265  interleave_permute_type = corenrn_param.cell_interleave_permute;
266  cellorder_nwarp = corenrn_param.nwarp;
267  use_solve_interleave = corenrn_param.cell_interleave_permute;
268 
269  if (corenrn_param.gpu && interleave_permute_type == 0) {
270  if (nrnmpi_myid == 0) {
271  printf(
272  " WARNING : GPU execution requires --cell-permute type 1 or 2. Setting it to 1.\n");
273  }
274  interleave_permute_type = 1;
275  use_solve_interleave = true;
276  }
277 
278  // multisend options
279  use_multisend_ = corenrn_param.multisend ? 1 : 0;
280  n_multisend_interval = corenrn_param.ms_subint;
281  use_phase2_ = (corenrn_param.ms_phases == 2) ? 1 : 0;
282 
283  // reading *.dat files and setting up the data structures, setting mindelay
284  nrn_setup(filesdat.c_str(),
285  is_mapping_needed,
286  checkPoints,
287  run_setup_cleanup,
288  corenrn_param.datpath.c_str(),
289  checkPoints.get_restore_path().c_str(),
290  &corenrn_param.mindelay);
291 
292  // Allgather spike compression and bin queuing.
293  nrn_use_bin_queue_ = corenrn_param.binqueue;
294  int spkcompress = corenrn_param.spkcompress;
295  nrnmpi_spike_compress(spkcompress, (spkcompress ? true : false), use_multisend_);
296 
297  if (!corenrn_param.is_quiet()) {
298  report_mem_usage("After nrn_setup ");
299  }
300 
301  // Invoke PatternStim
302  if (!corenrn_param.patternstim.empty()) {
303  nrn_mkPatternStim(corenrn_param.patternstim.c_str(), corenrn_param.tstop);
304  }
305 
306  /// Setting the timeout
307  nrn_set_timeout(200.);
308 
309  // show all configuration parameters for current run
310  if (nrnmpi_myid == 0 && !corenrn_param.is_quiet()) {
311  std::cout << corenrn_param << std::endl;
312  std::cout << " Start time (t) = " << t << std::endl << std::endl;
313  }
314 
315  // allocate buffer for mpi communication
316  mk_spikevec_buffer(corenrn_param.spikebuf);
317 
318  if (!corenrn_param.is_quiet()) {
319  report_mem_usage("After mk_spikevec_buffer");
320  }
321 
322  // In direct mode there are likely trajectory record requests
323  // to allow processing in NEURON after simulation by CoreNEURON
324  if (corenrn_embedded) {
325  // arg is additional vector size required (how many items will be
326  // written to the double*) but NEURON can instead
327  // specify that returns will be on a per time step basis.
328  get_nrn_trajectory_requests(int((corenrn_param.tstop - t) / corenrn_param.dt) + 2);
329 
330  // In direct mode, CoreNEURON has exactly the behavior of
331  // ParallelContext.psolve(tstop). Ie a sequence of such calls
332  // without an intervening h.finitialize() continues from the end
333  // of the previous call. I.e., all initial state, including
334  // the event queue has been set up in NEURON. And, at the end
335  // all final state, including the event queue will be sent back
336  // to NEURON. Here there is some first time only
337  // initialization and queue transfer.
338  direct_mode_initialize();
339  clear_spike_vectors(); // PreSyn send already recorded by NEURON
340  (*nrn2core_part2_clean_)();
341  }
342 
343  if (corenrn_param.gpu) {
344  // Copy nrnthreads to device only after all the data are passed from NEURON and the
345  // nrnthreads on CPU are properly set up
346  setup_nrnthreads_on_device(nrn_threads, nrn_nthread);
347  }
348 
349  if (corenrn_embedded) {
350  // Run nrn_init of mechanisms only to allocate any extra data needed on the GPU after
351  // nrnthreads are properly set up on the GPU
352  allocate_data_in_mechanism_nrn_init();
353  }
354 
355  if (nrn_have_gaps) {
356  nrn_partrans::gap_update_indices();
357  }
358 
359  // call prcellstate for prcellgid
360  call_prcellstate_for_prcellgid(corenrn_param.prcellgid, corenrn_param.gpu, 1);
361 }

388 /* perform forwardskip and call prcellstate for prcellgid */
389 void handle_forward_skip(double forwardskip, int prcellgid) {
390  double savedt = dt;
391  double savet = t;
392 
393  dt = forwardskip * 0.1;
394  t = -1e9;
395  dt2thread(-1.);
396 
397  for (int step = 0; step < 10; ++step) {
398  nrn_fixed_step_minimal();
399  }
400 
401  if (prcellgid >= 0) {
402  prcellstate(prcellgid, "fs");
403  }
404 
405  dt = savedt;
406  t = savet;
407  dt2thread(-1.);
408 
409  // clear spikes generated during forward skip (with negative time)
410  clear_spike_vectors();
411 }

518 extern "C" int run_solve_core(int argc, char** argv) {
519  Instrumentor::phase_begin("main");
520 
521  std::vector<ReportConfiguration> configs;
522  std::vector<std::unique_ptr<ReportHandler>> report_handlers;
523  std::vector<std::pair<std::string, int>> spikes_population_name_offset;
524  bool reports_needs_finalize = false;
525 
526  if (!corenrn_param.is_quiet()) {
527  report_mem_usage("After mk_mech");
528  }
529 
530  // Create outpath if it does not exist
531  if (nrnmpi_myid == 0) {
532  mkdir_p(corenrn_param.outpath.c_str());
533  }
534 
535  if (!corenrn_param.reportfilepath.empty()) {
536  configs = create_report_configurations(corenrn_param.reportfilepath,
537  corenrn_param.outpath,
538  spikes_population_name_offset);
539  reports_needs_finalize = !configs.empty();
540  }
541 
542  CheckPoints checkPoints{corenrn_param.checkpointpath, corenrn_param.restorepath};
543 
544  // initializationa and loading functions moved to separate
545  {
546  Instrumentor::phase p("load-model");
547  nrn_init_and_load_data(argc, argv, checkPoints, !configs.empty());
548  }
549 
550  std::string output_dir = corenrn_param.outpath;
551 
552  if (nrnmpi_myid == 0) {
553  mkdir_p(output_dir.c_str());
554  }
555 #if NRNMPI
556  if (corenrn_param.mpi_enable) {
557  nrnmpi_barrier();
558  }
559 #endif
560  bool compute_gpu = corenrn_param.gpu;
561 
562  // clang-format off
563 
564  #pragma acc update device(celsius, secondorder, pi) if (compute_gpu)
565  // clang-format on
566  {
567  double v = corenrn_param.voltage;
568  double dt = corenrn_param.dt;
569  double delay = corenrn_param.mindelay;
570  double tstop = corenrn_param.tstop;
571 
572  if (tstop < t && nrnmpi_myid == 0) {
573  printf("Error: Stop time (%lf) < Start time (%lf), restoring from checkpoint? \n",
574  tstop,
575  t);
576  abort();
577  }
578 
579  // TODO : if some ranks are empty then restore will go in deadlock
580  // phase (as some ranks won't have restored anything and hence return
581  // false in checkpoint_initialize
582  if (!corenrn_embedded && !checkPoints.initialize()) {
583  nrn_finitialize(v != 1000., v);
584  }
585 
586  if (!corenrn_param.is_quiet()) {
587  report_mem_usage("After nrn_finitialize");
588  }
589 
590  // register all reports into reportinglib
591  double min_report_dt = INT_MAX;
592  for (size_t i = 0; i < configs.size(); i++) {
593  std::unique_ptr<ReportHandler> report_handler = create_report_handler(configs[i]);
594  if (report_handler) {
595  report_handler->create_report(dt, tstop, delay);
596  report_handlers.push_back(std::move(report_handler));
597  }
598  if (configs[i].report_dt < min_report_dt) {
599  min_report_dt = configs[i].report_dt;
600  }
601  }
602  // Set the buffer size if is not the default value. Otherwise use report.conf on
603  // register_report
604  if (corenrn_param.report_buff_size != corenrn_param.report_buff_size_default) {
605  set_report_buffer_size(corenrn_param.report_buff_size);
606  }
607  setup_report_engine(min_report_dt, delay);
608  configs.clear();
609 
610  // call prcellstate for prcellgid
611  call_prcellstate_for_prcellgid(corenrn_param.prcellgid, compute_gpu, 0);
612 
613  // handle forwardskip
614  if (corenrn_param.forwardskip > 0.0) {
615  Instrumentor::phase p("handle-forward-skip");
616  handle_forward_skip(corenrn_param.forwardskip, corenrn_param.prcellgid);
617  }
618 
619  /// Solver execution
620  Instrumentor::start_profile();
621  Instrumentor::phase_begin("simulation");
622  BBS_netpar_solve(corenrn_param.tstop);
623  Instrumentor::phase_end("simulation");
624  Instrumentor::stop_profile();
625 
626  // update cpu copy of NrnThread from GPU
627  update_nrnthreads_on_host(nrn_threads, nrn_nthread);
628 
629  // direct mode and full trajectory gathering on CoreNEURON, send back.
630  if (corenrn_embedded) {
631  trajectory_return();
632  }
633 
634  // Report global cell statistics
635  if (!corenrn_param.is_quiet()) {
636  report_cell_stats();
637  }
638 
639  // prcellstate after end of solver
