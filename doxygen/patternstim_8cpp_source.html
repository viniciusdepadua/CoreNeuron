99 size_t read_raster_file(const char* fname, double** tvec, int** gidvec, double tstop) {
100  FILE* f = fopen(fname, "r");
101  nrn_assert(f);
102 
103  // skip first line containing "scatter" string
104  char dummy[100];
105  nrn_assert(fgets(dummy, 100, f));
106 
107  std::vector<std::pair<double, int>> spikes;
108  spikes.reserve(10000);
109 
110  double stime;
111  int gid;
112 
113  while (fscanf(f, "%lf %d\n", &stime, &gid) == 2) {
114  if (stime >= t && stime <= tstop) {
115  spikes.push_back(std::make_pair(stime, gid));
116  }
117  }
118 
119  fclose(f);
120 
121  // pattern.mod expects sorted spike raster (this is to avoid
122  // injecting all events at the begining of the simulation).
123  // sort spikes according to time
124  std::sort(spikes.begin(), spikes.end());
125 
126  // fill gid and time vectors
127  *tvec = (double*) emalloc(spikes.size() * sizeof(double));
128  *gidvec = (int*) emalloc(spikes.size() * sizeof(int));
129 
130  for (size_t i = 0; i < spikes.size(); i++) {
131  (*tvec)[i] = spikes[i].first;
132  (*gidvec)[i] = spikes[i].second;
133  }
134 
135  return spikes.size();
136 }

139 static NrnThreadMembList* alloc_nrn_thread_memb(int type) {
140  NrnThreadMembList* tml = (NrnThreadMembList*) emalloc(sizeof(NrnThreadMembList));
141  tml->dependencies = nullptr;
142  tml->ndependencies = 0;
143  tml->index = type;
144  tml->next = nullptr;
145 
146  // fill in tml->ml info. The data is not in the cache efficient
147  // NrnThread arrays but there should not be many of these instances.
148  int psize = corenrn.get_prop_param_size()[type];
149  int dsize = corenrn.get_prop_dparam_size()[type];
150  int layout = corenrn.get_mech_data_layout()[type];
151  tml->ml = (Memb_list*) emalloc(sizeof(Memb_list));
152  tml->ml->nodecount = 1;
153  tml->ml->_nodecount_padded = tml->ml->nodecount;
154  tml->ml->nodeindices = nullptr;
155  tml->ml->data = (double*) ecalloc(tml->ml->nodecount * psize, sizeof(double));
156  tml->ml->pdata = (Datum*) ecalloc(nrn_soa_padded_size(tml->ml->nodecount, layout) * dsize,
157  sizeof(Datum));
158  tml->ml->_thread = nullptr;
159  tml->ml->_net_receive_buffer = nullptr;
160  tml->ml->_net_send_buffer = nullptr;
161  tml->ml->_permute = nullptr;
162 
163  return tml;
164 }

173 Point_process* nrn_artcell_instantiate(const char* mechname) {
174  int type = nrn_get_mechtype(mechname);
175  NrnThread* nt = nrn_threads + 0;
176 
177  // printf("nrn_artcell_instantiate %s type=%d\n", mechname, type);
178 
179  // create and append to nt.tml
180  auto tml = alloc_nrn_thread_memb(type);
181 
182  assert(nt->_ml_list[type] == nullptr); // FIXME
183  nt->_ml_list[type] = tml->ml;
184 
185  if (!nt->tml) {
186  nt->tml = tml;
187  } else {
188  for (NrnThreadMembList* i = nt->tml; i; i = i->next) {
189  if (!i->next) {
190  i->next = tml;
191  break;
192  }
193  }
194  }
195 
196  // Here we have a problem with no easy general solution. ml->pdata are
197  // integer indexes into the nt->_data nt->_idata and nt->_vdata array
198  // depending on context,
199  // but nrn_setup.cpp allocated these to exactly have the size needed by
200  // the file defined model (at least for _vdata) and so there are no slots
201  // for pdata to index into for this new instance.
202  // So nrn_setup.cpp:phase2 needs to
203  // be notified that some extra space will be required. For now, defer
204  // the general situation of several instances for several types and
205  // demand that this method is never called more than once. We introduce
206  // a int nrn_extra_thread0_vdata (only that is needed by PatternStim)
207  // which will be used by
208  // nrn_setup.cpp:phase2 to allocate the appropriately larger
209  // _vdata arrays for thread 0 (without changing _nvdata so
210  // that we can fill in the indices here)
211  static int cnt = 0;
212  if (++cnt > 1) {
213  printf("nrn_artcell_instantiate cannot be called more than once\n");
214  assert(0);
215  }
216  // note that PatternStim internal usage for the 4 ppvar values is:
217  // #define _nd_area _nt->_data[_ppvar[0]] (not used since ARTIFICIAL_CELL)
218  // #define _p_ptr _nt->_vdata[_ppvar[2]] (the BBCORE_POINTER)
219  // #define _tqitem &(_nt->_vdata[_ppvar[3]]) (for net_send)
220  // and general external usage is:
221  // _nt->_vdata[_ppvar[1]] = Point_process*
222  //
223 
224  Point_process* pnt = new Point_process;
225  pnt->_type = type;
226  pnt->_tid = nt->id;
227  pnt->_i_instance = 0;
228  // as though all dparam index into _vdata
229  int dsize = corenrn.get_prop_dparam_size()[type];
230  assert(dsize <= nrn_extra_thread0_vdata);
231  for (int i = 0; i < dsize; ++i) {
232  tml->ml->pdata[i] = nt->_nvdata + i;
233  }
234  nt->_vdata[nt->_nvdata + 1] = (void*) pnt;
235 
236  return pnt;
237 }
