109  /** Parse a neuron section segment mapping
110  *
111  * Read count no of mappings for section to segment
112  */
113  template <typename T>
114  int read_mapping_info(T* mapinfo) {
115  int nsec, nseg, n_scan;
116  char line_buf[max_line_length], name[max_line_length];
117 
118  F.getline(line_buf, sizeof(line_buf));
119  n_scan = sscanf(line_buf, "%s %d %d", name, &nsec, &nseg);
120 
121  nrn_assert(n_scan == 3);
122 
123  mapinfo->name = std::string(name);
124 
125  if (nseg) {
126  std::vector<int> sec, seg;
127  sec.reserve(nseg);
128  seg.reserve(nseg);
129 
130  read_array<int>(&sec[0], nseg);
131  read_array<int>(&seg[0], nseg);
132 
133  for (int i = 0; i < nseg; i++) {
134  mapinfo->add_segment(sec[i], seg[i]);
135  }
136  }
137  return nseg;
138  }

140  /** Defined flag values for parse_array() */
141  enum parse_action { read, seek };

143  /** Generic parse function for an array of fixed length.
144  *
145  * \tparam T the array element type: may be \c int or \c double.
146  * \param p pointer to the target in memory for reading into.
147  * \param count number of items of type \a T to parse.
148  * \param action whether to validate and skip (\c seek) or
149  * copy array into memory (\c read).
150  * \return the supplied pointer value.
151  *
152  * Error if \a count is non-zero, \a flag is \c read, and
153  * the supplied pointer \p is null.
154  *
155  * Arrays are represented by a checkpoint line followed by
156  * the array items in increasing index order, in the native binary
157  * representation of the writing process.
158  */
159  template <typename T>
160  inline T* parse_array(T* p, size_t count, parse_action flag) {
161  if (count > 0 && flag != seek)
162  nrn_assert(p != 0);
163 
164  read_checkpoint_assert();
165  switch (flag) {
166  case seek:
167  F.seekg(count * sizeof(T), std::ios_base::cur);
168  break;
169  case read:
170  F.read((char*) p, count * sizeof(T));
171  break;
172  }
173 
174  nrn_assert(!F.fail());
175  return p;
176  }

212  /** Write a padded array. nb_elements is number of elements to write per line,
213  * line_width is full size of a line in nb elements**/
214  template <typename T>
215  void write_array(T* p,
216  size_t nb_elements,
217  size_t line_width,
218  size_t nb_lines,
219  bool to_transpose = false) {
220  nrn_assert(F.is_open());
221  nrn_assert(current_mode & std::ios::out);
222  write_checkpoint();
223  T* temp_cpy = new T[nb_elements * nb_lines];
224 
225  if (to_transpose) {
226  for (size_t i = 0; i < nb_lines; i++) {
227  for (size_t j = 0; j < nb_elements; j++) {
228  temp_cpy[i + j * nb_lines] = p[i * line_width + j];
229  }
230  }
231  } else {
232  memcpy(temp_cpy, p, nb_elements * sizeof(T) * nb_lines);
233  }
234  // AoS never use padding, SoA is translated above, so one write
235  // operation is enought in both cases
236  F.write((const char*) temp_cpy, nb_elements * sizeof(T) * nb_lines);
237  nrn_assert(!F.fail());
238  delete[] temp_cpy;
239  }
