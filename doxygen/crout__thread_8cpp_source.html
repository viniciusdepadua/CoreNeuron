12 /* CROUT */

14 /* Performs an LU triangular factorization of a real matrix */
15 /* by the Crout algorithm using partial pivoting. Rows are */
16 /* not normalized; implicit equilibration is used. ROUNDOFF */
17 /* is the minimal value for a pivot element without its being*/
18 /* considered too close to zero (currently set to 1.0E-20). */
19 /* */
20 /* Returns: 0 if no error; 2 if matrix is singular or ill- */
21 /* conditioned */
22 /* */
23 /* Calling sequence: crout(n, a, perm) */
24 /* */
25 /* Arguments: */
26 /* */
27 /* Input: n, integer, number of rows of the matrix */
28 /* */
29 /* a, double precision matrix to be factored */
30 /* */
31 /* Output: */
32 /* */
33 /* a, factors required to transform the constant */
34 /* vector in the set of simultaneous equations */
35 /* are stored in the lower triangle; factors for */
36 /* back substitution are stored in the upper */

53 int nrn_crout_thread(NewtonSpace* ns, int n, double** a, int* perm, _threadargsproto_) {
54  int save_i = 0;
55 
56  /* Initialize permutation and rowmax vectors */
57 
58  double* rowmax = ns->rowmax;
59  for (int i = 0; i < n; i++) {
60  perm[ix(i)] = i;
61  int k = 0;
62  for (int j = 1; j < n; j++)
63  if (fabs(a[i][ix(j)]) > fabs(a[i][ix(k)]))
64  k = j;
65  rowmax[ix(i)] = a[i][ix(k)];
66  }
67 
68  /* Loop over rows and columns r */
69 
70  for (int r = 0; r < n; r++) {
71  /*
72  * Operate on rth column. This produces the lower triangular matrix
73  * of terms needed to transform the constant vector.
74  */
75 
76  for (int i = r; i < n; i++) {
77  double sum = 0.0;
78  int irow = perm[ix(i)];
79  for (int k = 0; k < r; k++) {
80  int krow = perm[ix(k)];
81  sum += a[irow][ix(k)] * a[krow][ix(r)];
82  }
83  a[irow][ix(r)] -= sum;
84  }
85 
86  /* Find row containing the pivot in the rth column */
87 
88  int pivot = perm[ix(r)];
89  double equil_1 = fabs(a[pivot][ix(r)] / rowmax[ix(pivot)]);
90  for (int i = r + 1; i < n; i++) {
91  int irow = perm[ix(i)];
92  double equil_2 = fabs(a[irow][ix(r)] / rowmax[ix(irow)]);
93  if (equil_2 > equil_1) {
94  /* make irow the new pivot row */
95 
96  pivot = irow;
97  save_i = i;
98  equil_1 = equil_2;
99  }
100  }
101 
102  /* Interchange entries in permutation vector if necessary */
103 
104  if (pivot != perm[ix(r)]) {
105  perm[ix(save_i)] = perm[ix(r)];
106  perm[ix(r)] = pivot;
107  }
108 
109  /* Check that pivot element is not too small */
110 
111  if (fabs(a[pivot][ix(r)]) < ROUNDOFF)
112  return (SINGULAR);
113 
114  /*
115  * Operate on row in rth position. This produces the upper
116  * triangular matrix whose diagonal elements are assumed to be unity.
117  * This matrix is used in the back substitution algorithm.
118  */
119 
120  for (int j = r + 1; j < n; j++) {
121  double sum = 0.0;
122  for (int k = 0; k < r; k++) {
123  int krow = perm[ix(k)];
124  sum += a[pivot][ix(k)] * a[krow][ix(j)];
125  }
126  a[pivot][ix(j)] = (a[pivot][ix(j)] - sum) / a[pivot][ix(r)];
127  }
128  }
129  return (SUCCESS);
130 }
