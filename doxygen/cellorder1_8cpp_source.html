28  * \brief Function to order trees by size, hash and nodeindex
29  */
30 static bool tnode_earlier(TNode* a, TNode* b) {
31  bool result = false;
32  if (a->treesize < b->treesize) { // treesize dominates
33  result = true;
34  } else if (a->treesize == b->treesize) {
35  if (a->hash < b->hash) { // if treesize same, keep identical trees together
36  result = true;
37  } else if (a->hash == b->hash) {
38  result = a->nodeindex < b->nodeindex; // identical trees ordered by nodeindex
39  }
40  }
41  return result;
42 }

63 size_t TNode::mkhash() { // call on all nodes in leaf to root order
64  // concept from http://stackoverflow.com/questions/20511347/a-good-hash-function-for-a-vector
65  std::sort(children.begin(), children.end(), ptr_tnode_earlier);
66  hash = children.size();
67  treesize = 1;
68  for (size_t i = 0; i < children.size(); ++i) { // need sorted by child hash
69  hash ^= children[i]->hash + 0x9e3779b9 + (hash << 6) + (hash >> 2);
70  treesize += children[i]->treesize;
71  }
72  return hash; // hash of leaf nodes is 0
73 }

120 /*
121 assess the quality of the ordering. The measure is the size of a contiguous
122 list of nodes whose parents have the same order. How many contiguous lists
123 have that same size. How many nodes participate in that size list.
124 Modify the quality measure from experience with performance. Start with
125 list of (nnode, size_participation)
126 */
127 static void quality(VecTNode& nodevec, size_t max = 32) {
128  size_t qcnt = 0; // how many contiguous nodes have contiguous parents
129 
130  // first ncell nodes are by definition in contiguous order
131  for (const auto& n: nodevec) {
132  if (n->parent != nullptr) {
133  break;
134  }
135  qcnt += 1;
136  }
137  size_t ncell = qcnt;
138 
139  // key is how many parents in contiguous order
140  // value is number of nodes that participate in that
141  std::map<size_t, size_t> qual;
142  size_t ip_last = 10000000000;
143  for (size_t i = ncell; i < nodevec.size(); ++i) {
144  size_t ip = nodevec[i]->parent->nodevec_index;
145  // i%max == 0 means that if we start a warp with 8 and then have 32
146  // the 32 is broken into 24 and 8. (modify if the arrangement during
147  // gaussian elimination becomes more sophisticated.(
148  if (ip == ip_last + 1 && i % max != 0) { // contiguous
149  qcnt += 1;
150  } else {
151  if (qcnt == 1) {
152  // printf("unique %ld p=%ld ix=%d\n", i, ip, nodevec[i]->nodeindex);
153  }
154  qual[max] += (qcnt / max) * max;
155  size_t x = qcnt % max;
156  if (x) {
157  qual[x] += x;
158  }
159  qcnt = 1;
160  }
161  ip_last = ip;
162  }
163  qual[max] += (qcnt / max) * max;
164  size_t x = qcnt % max;
165  if (x) {
166  qual[x] += x;
167  }
168 
169  // print result
170  qcnt = 0;
171 #if CORENRN_DEBUG
172  for (const auto& q: qual) {
173  qcnt += q.second;
174  printf("%6ld %6ld\n", q.first, q.second);
175  }
176 #endif
177 #if CORENRN_DEBUG
178  printf("qual.size=%ld qual total nodes=%ld nodevec.size=%ld\n",
179  qual.size(),
180  qcnt,
181  nodevec.size());
182 #endif
183 
184  // how many race conditions. ie refer to same parent on different core
185  // of warp (max cores) or parent in same group of max.
186  size_t maxip = ncell;
187  size_t nrace1 = 0;
188  size_t nrace2 = 0;
189  std::set<size_t> ipused;
190  for (size_t i = ncell; i < nodevec.size(); ++i) {
191  TNode* nd = nodevec[i];
192  size_t ip = nd->parent->nodevec_index;
193  if (i % max == 0) {
194  maxip = i;
195  ipused.clear();
196  }
197  if (ip >= maxip) {
198  nrace1 += 1;
199  } /*else*/
200  {
201  if (ipused.find(ip) != ipused.end()) {
202  nrace2 += 1;
203  if (ip >= maxip) {
204  // printf("race for parent %ld (parent in same group as multiple users))\n",
205  // ip);
206  }
207  } else {
208  ipused.insert(ip);
209  }
210  }
211  }
212 #if CORENRN_DEBUG
213  printf("nrace = %ld (parent in same group of %ld nodes)\n", nrace1, max);
214  printf("nrace = %ld (parent used more than once by same group of %ld nodes)\n", nrace2, max);
215 #endif
216 }

222  nd->level = nd->parent->level + 1;

233 size_t level_from_leaf(VecTNode& nodevec) {
234  size_t maxlevel = 0;
235  for (size_t i = nodevec.size() - 1; true; --i) {
236  TNode* nd = nodevec[i];
237  size_t lmax = 0;
238  for (auto& child: nd->children) {
239  if (lmax <= child->level) {
240  lmax = child->level + 1;
241  }
242  }
243  nd->level = lmax;
244  if (maxlevel < lmax) {
245  maxlevel = lmax;
246  }
247  if (i == 0) {
248  break;
249  }
250  }
251  return maxlevel;

435  * \brief Perform tree preparation for interleaving strategies
436  *
437  * \param parent vector of parent indices
438  * \param nnode number of compartments in the cells
439  * \param ncell number of cells
440  */
441 void tree_analysis(int* parent, int nnode, int ncell, VecTNode& nodevec) {
442  // create empty TNodes (knowing only their index)
443  nodevec.reserve(nnode);
444  for (int i = 0; i < nnode; ++i) {
445  nodevec.push_back(new TNode(i));
446  }
447 
448  // determine the (sorted by hash) children of each node
449  for (int i = nnode - 1; i >= ncell; --i) {
450  nodevec[i]->parent = nodevec[parent[i]];
451  nodevec[i]->mkhash();
452  nodevec[parent[i]]->children.push_back(nodevec[i]);
453  }
454 
455  // determine hash of the cells
456  for (int i = 0; i < ncell; ++i) {
457  nodevec[i]->mkhash();
458  }
459 
460  // sort it by tree size (from smaller to larger)
461  std::sort(nodevec.begin(), nodevec.begin() + ncell, tnode_earlier);
462 }

476 /**
477  * \brief Naive interleaving strategy (interleave_permute_type == 1)
478  *
479  * Sort so nodevec[ncell:nnode] cell instances are interleaved. Keep the
480  * secondary ordering with respect to treenode_order so each cell is still a tree.
481  *
482  * \param ncell number of cells (trees)
483  * \param nodevec vector that contains compartments (nodes of the trees)
484  */
485 void node_interleave_order(int ncell, VecTNode& nodevec) {
486  int* order = new int[ncell];
487  for (int i = 0; i < ncell; ++i) {
488  order[i] = 0;
489  nodevec[i]->treenode_order = order[i]++;
490  }
491  for (size_t i = 0; i < nodevec.size(); ++i) {
492  TNode& nd = *nodevec[i];
493  for (size_t j = 0; j < nd.children.size(); ++j) {
494  TNode* cnode = nd.children[j];
495  cnode->treenode_order = order[nd.cellindex]++;
496  }
497  }
498  delete[] order;
499 
500  // std::sort(nodevec.begin() + ncell, nodevec.end(), contig_comp);
501  // Traversal of nodevec: From root to leaves (this is why we compute the tree node order)
502  std::sort(nodevec.begin() + ncell, nodevec.end(), interleave_comp);
503 
504 #if CORENRN_DEBUG
505  for (size_t i = 0; i < nodevec.size(); ++i) {
506  TNode& nd = *nodevec[i];
507  printf("%ld cell=%ld ix=%d\n", i, nd.cellindex, nd.nodeindex);
508  }
509 #endif
510 }

512 static void admin1(int ncell,
513  VecTNode& nodevec,
514  int& nwarp,
515  int& nstride,
516  int*& stride,
517  int*& firstnode,
518  int*& lastnode,
519  int*& cellsize) {
520  firstnode = (int*) ecalloc_align(ncell, sizeof(int));
521  lastnode = (int*) ecalloc_align(ncell, sizeof(int));
522  cellsize = (int*) ecalloc_align(ncell, sizeof(int));
523 
524  nwarp = (ncell % warpsize == 0) ? (ncell / warpsize) : (ncell / warpsize + 1);
525 
526  for (int i = 0; i < ncell; ++i) {
527  firstnode[i] = -1;
528  lastnode[i] = -1;
529  cellsize[i] = 0;
530  }
531 
532  nstride = 0;
533  for (size_t i = ncell; i < nodevec.size(); ++i) {
534  TNode& nd = *nodevec[i];
535  size_t ci = nd.cellindex;
536  if (firstnode[ci] == -1) {
537  firstnode[ci] = i;
538  }
539  lastnode[ci] = i;
540  cellsize[ci] += 1;
541  if (nstride < cellsize[ci]) {
542  nstride = cellsize[ci];
543  }
544  }
545 
546  stride = (int*) ecalloc_align(nstride + 1, sizeof(int));
547  for (int i = 0; i <= nstride; ++i) {
548  stride[i] = 0;
549  }
550  for (size_t i = ncell; i < nodevec.size(); ++i) {
551  TNode& nd = *nodevec[i];
552  stride[nd.treenode_order - 1] += 1; // -1 because treenode order includes root
553  }

556 // for admin2 we allow the node organisation in warps of (say 4 cores per warp)
557 // ............... ideal warp but unbalanced relative to warp with max cycles
558 // ............... ncycle = 15, icore [0:4), all strides are 4.
559 // ...............
560 // ...............
561 //
562 // .......... unbalanced relative to warp with max cycles
563 // .......... ncycle = 10, not all strides the same because
564 // .......... of need to avoid occasional race conditions.
565 // . . .. icore [4:8) only 4 strides of 4
566 //
567 // .................... ncycle = 20, uses only one core in the warp (cable)
568 // icore 8, all ncycle strides are 1
569 
570 // One thing to be unhappy about is the large stride vector of size about
571 // number of compartments/warpsize. There are a lot of models where the
572 // stride for a warp is constant except for one cycle in the warp and that
573 // is easy to obtain when there are more than warpsize cells per warp.

575 static size_t stride_length(size_t begin, size_t end, VecTNode& nodevec) {
576  // return stride length starting at i. Do not go past j.
577  // max stride is warpsize.
578  // At this time, only assume vicious parent race conditions matter.
579  if (end - begin > warpsize) {
580  end = begin + warpsize;
581  }
582  for (size_t i = begin; i < end; ++i) {
583  TNode* nd = nodevec[i];
584  nrn_assert(nd->nodevec_index == i);
585  size_t diff = dist2child(nd);
586  if (i + diff < end) {
587  end = i + diff;
588  }
589  }
590  return end - begin;
591 }

593 static void admin2(int ncell,
594  VecTNode& nodevec,
595  int& nwarp,
596  int& nstride,
597  int*& stridedispl,
598  int*& strides,
599  int*& rootbegin,
600  int*& nodebegin,
601  int*& ncycles) {
602  // the number of groups is the number of warps needed
603  // ncore is the number of warps * warpsize
604  nwarp = nodevec[ncell - 1]->groupindex + 1;
605 
606  ncycles = (int*) ecalloc_align(nwarp, sizeof(int));
607  stridedispl = (int*) ecalloc_align(nwarp + 1,
608  sizeof(int)); // running sum of ncycles (start at 0)
609  rootbegin = (int*) ecalloc_align(nwarp + 1, sizeof(int)); // index (+1) of first root in warp.
610  nodebegin = (int*) ecalloc_align(nwarp + 1, sizeof(int)); // index (+1) of first node in warp.
611 
612  // rootbegin and nodebegin are the root index values + 1 of the last of
613  // the sequence of constant groupindex
614  rootbegin[0] = 0;
615  for (size_t i = 0; i < size_t(ncell); ++i) {
616  rootbegin[nodevec[i]->groupindex + 1] = i + 1;
617  }
618  nodebegin[0] = ncell;
619  // We start from the leaves and go backwards towards the root
620  for (size_t i = size_t(ncell); i < nodevec.size(); ++i) {
621  nodebegin[nodevec[i]->groupindex + 1] = i + 1;
622  }
623 
624  // ncycles, stridedispl, and nstride
625  nstride = 0;
626  stridedispl[0] = 0;
627  for (size_t iwarp = 0; iwarp < (size_t) nwarp; ++iwarp) {
628  size_t j = size_t(nodebegin[iwarp + 1]);
629  int nc = 0;
630  size_t i = nodebegin[iwarp];
631  while (i < j) {
632  i += stride_length(i, j, nodevec);
633  ++nc; // how many times the warp should loop in order to finish with all the tree
634  // depths (for all the trees of the warp/group)
635  }
636  ncycles[iwarp] = nc;
637  stridedispl[iwarp + 1] = stridedispl[iwarp] + nc;
638  nstride += nc;
639  }
640 
641  // strides
642  strides = (int*) ecalloc_align(nstride, sizeof(int));
643  nstride = 0;
644  for (size_t iwarp = 0; iwarp < (size_t) nwarp; ++iwarp) {
645  size_t j = size_t(nodebegin[iwarp + 1]);
646  size_t i = nodebegin[iwarp];
647  while (i < j) {
648  int k = stride_length(i, j, nodevec);
649  i += k;
650  strides[nstride++] = k;
651  }
652  }
