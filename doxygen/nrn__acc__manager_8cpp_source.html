557 static void net_receive_buffer_order(NetReceiveBuffer_t* nrb) {
558  Instrumentor::phase p_net_receive_buffer_order("net-receive-buf-order");
559  if (nrb->_cnt == 0) {
560  nrb->_displ_cnt = 0;
561  return;
562  }
563 
564  std::priority_queue<NRB_P, std::vector<NRB_P>, comp> nrbq;
565 
566  for (int i = 0; i < nrb->_cnt; ++i) {
567  nrbq.push(NRB_P(nrb->_pnt_index[i], i));
568  }
569 
570  int displ_cnt = 0;
571  int index_cnt = 0;
572  int last_instance_index = -1;
573  nrb->_displ[0] = 0;
574 
575  while (!nrbq.empty()) {
576  const NRB_P& p = nrbq.top();
577  nrb->_nrb_index[index_cnt++] = p.second;
578  if (p.first != last_instance_index) {
579  ++displ_cnt;
580  }
581  nrb->_displ[displ_cnt] = index_cnt;
582  last_instance_index = p.first;
583  nrbq.pop();
584  }
585  nrb->_displ_cnt = displ_cnt;
586 }
