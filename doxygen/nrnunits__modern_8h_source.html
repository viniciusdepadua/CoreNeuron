9 #ifndef nrnunits_modern_h
10 #define nrnunits_modern_h
11 
12 /**
13  NMODL translated MOD files get unit constants typically from
14  share/lib/nrnunits.lib.in. But there were other source files that
15  hardcode some of the constants. Here we gather a few modern units into
16  a single place (but, unfortunately, also in nrnunits.lib.in). Legacy units
17  cannot be gathered here because they can differ slightly from place to place.
18 
19  These come from https://physics.nist.gov/cuu/Constants/index.html.
20  Termed the "2018 CODATA recommended values", they became available
21  on 20 May 2019 and replace the 2014 CODATA set.
22 
23  See oc/hoc_init.c, nrnoc/eion.c, nrniv/kschan.h
24 **/
25 
26 
27 #define _electron_charge_codata2018 1.602176634e-19 /* coulomb exact*/
28 #define _avogadro_number_codata2018 6.02214076e+23 /* exact */
29 #define _boltzmann_codata2018 1.380649e-23 /* joule/K exact */
30 #define _faraday_codata2018 \
31  (_electron_charge_codata2018 * _avogadro_number_codata2018) /* 96485.33212... coulomb/mol */
32 #define _gasconstant_codata2018 \
33  (_boltzmann_codata2018 * _avogadro_number_codata2018) /* 8.314462618... joule/mol-K */
34 
35 /* e/k in K/millivolt */
36 #define _e_over_k_codata2018 \
37  (.001 * _electron_charge_codata2018 / _boltzmann_codata2018) /* 11.604518... K/mV */
38 
39 #endif /* nrnunits_modern_h */
