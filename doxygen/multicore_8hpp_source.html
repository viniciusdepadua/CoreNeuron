158  int i;
159  // clang-format off
160 
161  #pragma omp parallel for private(i) shared(nrn_threads, job, nrn_nthread, \
162  nrnmpi_myid) schedule(static, 1)
163  for (i = 0; i < nrn_nthread; ++i) {
164  job(nrn_threads + i, std::forward<Args>(args)...);
165  }
166  // clang-format on
